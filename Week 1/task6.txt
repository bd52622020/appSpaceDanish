def ispalin(word):
    word2 = word[::-1]
    if word == word2:
        print("this string is a palindrome")
    else:
        print("this string is not a palindrome")

import re 

def match(string):
    pattern = '^[a-zA-Z0-9_]*$'
    if re.search(pattern, string):
        print("match found")
    else:
        print("no match found")
